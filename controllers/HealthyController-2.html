import { Controller, Get } from '@nestjs/common';

@Controller('healthy')
export class HealthyController {
  private readonly logger = new CastLogger(
    HealthyController.name,
    CastLoggerOptions
  );

  @Get()
  getData() {
    this.logger.log('Health Check');
    return '';
  }
}
